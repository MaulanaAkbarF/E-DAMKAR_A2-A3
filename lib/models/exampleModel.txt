import 'dart:convert';
import 'package:http/http.dart' as http;

const String _baseUrl = 'http://188.10.10.254:8080/belajarphp/';

class LoginModel {
  static var client = http.Client();

  String name;
  String job;
  String id;
  String createdAt;

  LoginModel(
      {required this.name,
      required this.job,
      required this.id,
      required this.createdAt});

  factory LoginModel.createResult(Map<String, dynamic> object) {
    return LoginModel(
        name: object['name'],
        job: object['job'],
        id: object['id'],
        createdAt: object['createdAt']);
  }

  static Future<LoginModel> postData(String username, String pass) async {
    String url = 'https://reqres.in/api/users';
    Uri ur = Uri.parse(url);
    var result = await http.post(ur, body: {"name": username, "job": pass});
    var jsonobj = json.decode(result.body);
    return LoginModel.createResult(jsonobj);
  }
}
